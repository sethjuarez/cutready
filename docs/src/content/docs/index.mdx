<WorkflowSteps>
  <WorkflowStep title="Sketch">
    Plan your demo with a rich editor — add narrative, actions, and reference screenshots.
  </WorkflowStep>
  <WorkflowStep title="Record">
    Capture browser interactions automatically with Playwright-powered recording.
  </WorkflowStep>
  <WorkflowStep title="Refine">
    AI agent cleans up actions, generates narration, and stabilizes selectors.
  </WorkflowStep>
  <WorkflowStep title="Produce">
    Replay and record lossless video with synchronized audio tracks.
  </WorkflowStep>
  <WorkflowStep title="Export">
    Package into FCPXML timelines for DaVinci Resolve or other editors.
  </WorkflowStep>
</WorkflowSteps>
